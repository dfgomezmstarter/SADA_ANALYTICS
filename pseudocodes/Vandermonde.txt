Input: array X, array Y 
Output: array Coef

int n <- X.size()
matrix A <- zeros_matrix(n, n)

for (int i from 0 until n)
	for (int j from 0 until n)
		A[j][i] <- X[j]^(n - (i + 1))
	end for
end for

array coef <- solution(A, Y.transposed) // solution is a function that solve systems of equations

return coef
